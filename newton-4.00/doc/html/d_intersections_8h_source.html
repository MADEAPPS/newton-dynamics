80 D_INLINE void dMovingAABB (dVector& p0, dVector& p1, const dVector& veloc, const dVector& omega, dFloat32 timestep, dFloat32 maxRadius, dFloat32 minRadius)
81 {
82  dVector linearStep (veloc.Scale (timestep));
83 
84  // estimate the maximum effect of the angular velocity and enlarge that box by that value (use 45 degrees as max angle not 90)
85  dAssert (omega.m_w == dFloat32 (0.0f));
86  dFloat32 maxAngle = dMin (dSqrt (omega.DotProduct(omega).GetScalar() * timestep * timestep), dFloat32 (45.0f * dDegreeToRad));
87 
88  dFloat32 angularTravel = (maxRadius - minRadius) * maxAngle;
89  dVector angularStep (angularTravel, angularTravel, angularTravel, dFloat32 (0.0f));
90 
91  dVector r0 (p0 - angularStep);
92  dVector r1 (p1 + angularStep);
93  dVector q0 (r0 + linearStep);
94  dVector q1 (r1 + linearStep);
95  p0 = r0.GetMin (q0) & dVector::m_triplexMask;
96  p1 = r1.GetMax (q1) & dVector::m_triplexMask;
97 }

99 D_INLINE dFloat32 dBoxPenetration (const dVector& minBox, const dVector& maxBox)
100 {
101  dAssert(maxBox.m_x >= minBox.m_x);
102  dAssert(maxBox.m_y >= minBox.m_y);
103  dAssert(maxBox.m_z >= minBox.m_z);
104 
105  dVector mask ((minBox * maxBox) < dVector::m_zero);
106  dVector dist (maxBox.GetMin (minBox.Abs()) & mask);
107  dist = dist.GetMin(dist.ShiftTripleRight());
108  dist = dist.GetMin(dist.ShiftTripleRight());
109  return dist.GetScalar();
110 }

146  D_INLINE dInt32 BoxTest(const dVector& minBox, const dVector& maxBox) const
147  {
148 #if 1
149  dVector test(((m_p0 <= minBox) | (m_p0 >= maxBox)) & m_isParallel);
150  if (test.GetSignMask() & 0x07) {
151  return 0;
152  }
153 
154  dVector tt0(m_dpInv * (minBox - m_p0));
155  dVector tt1(m_dpInv * (maxBox - m_p0));
156 
157  dVector t0(m_minT.GetMax(tt0.GetMin(tt1)));
158  dVector t1(m_maxT.GetMin(tt0.GetMax(tt1)));
159  t0 = t0.GetMax(t0.ShiftTripleRight());
160  t1 = t1.GetMin(t1.ShiftTripleRight());
161  t0 = t0.GetMax(t0.ShiftTripleRight());
162  t1 = t1.GetMin(t1.ShiftTripleRight());
163  return ((t0 < t1).GetSignMask() & 1);
164 
165 #else
166 
167  dFloat32 tmin = 0.0f;
168  dFloat32 tmax = 1.0f;
169 
170  for (dInt32 i = 0; i < 3; i++)
171  {
172  if (m_isParallel[i])
173  {
174  if (m_p0[i] <= minBox[i] || m_p0[i] >= maxBox[i])
175  {
176  return 0;
177  }
178  }
179  else
180  {
181  dFloat32 t1 = (minBox[i] - m_p0[i]) * m_dpInv[i];
182  dFloat32 t2 = (maxBox[i] - m_p0[i]) * m_dpInv[i];
183 
184  if (t1 > t2)
185  {
186  dSwap(t1, t2);
187  }
188  if (t1 > tmin)
189  {
190  tmin = t1;
191  }
192  if (t2 < tmax)
193  {
194  tmax = t2;
195  }
196  if (tmin > tmax)
197  {
198  return 0;
199  }
200  }
201  }
202  return 0x1;
203 #endif
204  }

206  D_INLINE dFloat32 BoxIntersect(const dVector& minBox, const dVector& maxBox) const
207  {
208  dVector test(((m_p0 <= minBox) | (m_p0 >= maxBox)) & m_isParallel);
209  if (test.GetSignMask() & 0x07)
210  {
211  return dFloat32(1.2f);
212  }
213  dVector tt0(m_dpInv * (minBox - m_p0));
214  dVector tt1(m_dpInv * (maxBox - m_p0));
215  dVector t0(m_minT.GetMax(tt0.GetMin(tt1)));
216  dVector t1(m_maxT.GetMin(tt0.GetMax(tt1)));
217  t0 = t0.GetMax(t0.ShiftTripleRight());
218  t1 = t1.GetMin(t1.ShiftTripleRight());
219  t0 = t0.GetMax(t0.ShiftTripleRight());
220  t1 = t1.GetMin(t1.ShiftTripleRight());
221  dVector mask(t0 < t1);
222  dVector maxDist(dFloat32(1.2f));
223  t0 = maxDist.Select(t0, mask);
224  dAssert((mask.GetSignMask() & 1) == (t0.m_x < dFloat32(1.0f)));
225  return t0.GetScalar();
226  }
