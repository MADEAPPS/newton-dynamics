205 inline void ndPolygonMeshDesc::SetDistanceTravel(const dVector& distanceInGlobalSpace)
206 {
207  const dMatrix& soupMatrix = m_polySoupInstance->GetGlobalMatrix();
208  m_boxDistanceTravelInMeshSpace = m_polySoupInstance->GetInvScale() * soupMatrix.UnrotateVector(distanceInGlobalSpace * m_convexInstance->GetInvScale());
209  if (m_boxDistanceTravelInMeshSpace.DotProduct(m_boxDistanceTravelInMeshSpace).GetScalar() < dFloat32(1.0e-2f))
210  {
211  m_doContinuesCollisionTest = false;
212  }
213 }
